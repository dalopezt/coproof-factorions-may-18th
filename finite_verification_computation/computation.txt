def run(input_data, target):
    import math
    fact = [math.factorial(d) for d in range(10)]
    expected = set(target["expected_factorions"])
    evidence = []
    all_found = []
    for chunk in input_data:
        start = chunk["start"]
        end = chunk["end"]
        found = []
        for n in range(start, end + 1):
            s = 0
            tmp = n
            while tmp > 0:
                s += fact[tmp % 10]
                tmp //= 10
            if s == n:
                found.append(n)
        chunk_sufficient = all(f in expected for f in found)
        evidence.append({"chunk_start": start, "chunk_end": end, "found": found, "sufficient": chunk_sufficient})
        all_found.extend(found)
    overall_sufficient = all(f in expected for f in all_found)
    return {
        "evidence": evidence,
        "sufficient": overall_sufficient,
        "summary": "Scanned 1..2540160 in 3 chunks. Factorions found: " + str(all_found),
        "records": all_found
    }
