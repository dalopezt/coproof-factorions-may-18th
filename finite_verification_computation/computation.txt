def run(input_data, target):
    import math
    fact = [math.factorial(d) for d in range(10)]
    expected = set(target["expected_factorions"])
    all_found = []
    evidence = []
    for chunk in input_data:
        start = chunk["start"]
        end = chunk["end"]
        found = []
        for n in range(start, end + 1):
            s = 0
            tmp = n
            while tmp > 0:
                s += fact[tmp % 10]
                tmp //= 10
            if s == n:
                found.append(n)
        chunk_sufficient = all(n in expected for n in found)
        all_found.extend(found)
        evidence.append({"chunk_start": start, "chunk_end": end, "found": found, "sufficient": chunk_sufficient})
    sufficient = all(ev["sufficient"] for ev in evidence)
    return {"evidence": evidence, "sufficient": sufficient, "summary": "Scanned " + str(len(evidence)) + " chunk(s) across ranks; factorions found: " + str(all_found), "records": all_found}
